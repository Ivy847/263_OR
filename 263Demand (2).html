import numpy as np

# Overestimation calculation dictionary
overestimation = {}
overestimation_mean = {}
overestimation_total = {}

for store in weekdays_indexed.index.unique():
    differences = []
    for boxes in weekdays_indexed.loc[store].values:
        if boxes < percentiles_stores[store]:
            differences.append(percentiles_stores[store] - boxes)
    overestimation[store] = differences
    overestimation_mean[store] = np.mean(differences)
    overestimation_total[store] = np.sum(differences)

# Print total overestimation for each store
for store, total in overestimation_total.items():
    print(store, ":", total)

avg_series = pd.Series(overestimation_mean).sort_values()
